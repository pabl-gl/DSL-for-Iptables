# Simple firewall configuration
# Comments are supported!

// Allow SSH from local network
allow from 192.168.1.0/24 to any proto tcp port 4321243244

// Allow established connections
accept from any to any when state ESTABLISHED

// Allow HTTP and HTTPS
allow proto tcp port 80 from any to any
allow proto tcp port 443 from any to any

// Block specific IP
block from 10.0.256.1/2423 to any

// Allow all from localhost
allow from 127.0.0.1

// Port ranges work too
allow from any to any proto tcp port 8000-8

// Drop everything else by default
drop from default
